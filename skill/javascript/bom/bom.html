# 进阶封装

    _toSetStyle(el,attrs){
        if(!el) return
        for(let key in attrs){
            let val = attrs[key]
            if(key === 'transform'){
            el.style[key] = val
            }else if (key === 'opacity'){
            el.style[key] = val
            }else{
            el.style[key] = `${this.scale * val}px`
            }
        }
    }
    //使用
    this._toSetStyle(this.$refs['swiperContainer'],{
        width:700,
        height:686,
        opacity:1
    })
